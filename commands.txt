AWS Scalable Web App Project - Windows Notes

# Connect to EC2 using PEM file

ssh -i "your-key.pem" ubuntu@<PUBLIC-IP>

# Apache Commands (Ubuntu EC2)

sudo apt update -y

sudo apt install apache2 -y

sudo systemctl start apache2

sudo systemctl enable apache2

sudo systemctl status apache2

# Create Web Page

echo "Server 1" | sudo tee /var/www/html/index.html

# Verify Web Page

curl localhost

# Check Port 80

sudo ss -tulpn | grep :80

# Change Page Content

echo "Server 2" | sudo tee /var/www/html/index.html

echo "Server 3" | sudo tee /var/www/html/index.html

# Check HTTP Response

curl -I http://localhost

# Auto Scaling Test

yes > /dev/null

# Stop Test

Ctrl + C
